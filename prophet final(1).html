from sklearn.metrics import mean_absolute_error, mean_squared_error, r2_score
import numpy as np
# Merge forecast with actuals
merged = forecast[['ds', 'yhat']].merge(weekly_sales, on='ds', how='left')

# Drop any rows where actuals are missing (e.g., future dates)
eval_df = merged.dropna(subset=['y'])

# Extract predictions and true values
y_true = eval_df['y']
y_pred = eval_df['yhat']

# Calculate metrics
mae = mean_absolute_error(y_true, y_pred)
mse = mean_squared_error(y_true, y_pred)
rmse = np.sqrt(mse)
r2 = r2_score(y_true, y_pred)
relative_error = (mae / np.mean(y_true)) * 100
pseudo_accuracy = 100 - relative_error

print(f"Mean Absolute Error (MAE): {mae:.2f}")
print(f"Mean Squared Error (MSE): {mse:.2f}")
print(f"Root Mean Squared Error (RMSE): {rmse:.2f}")
print(f"R² Score: {r2:.2f}")
print(f"Relative Error: {relative_error:.2f}%")
print(f"Pseudo-Accuracy (100 - Relative Error): {pseudo_accuracy:.2f}%")
